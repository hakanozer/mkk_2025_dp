Domain Model ve Builder Interface

@Entity
@Table(name = "api_configurations")
public final class ApiConfiguration {
    
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private final Long id;
    
    @Column(nullable = false)
    private final String serviceName;
    
    @Column(nullable = false)
    private final String baseUrl;
    
    private final String apiVersion;
    private final int timeoutMs;
    private final int retryCount;
    private final boolean sslEnabled;
    private final String authToken;
    private final String userAgent;
    
    @ElementCollection
    @CollectionTable(name = "api_headers")
    private final Map<String, String> customHeaders;
    
    // Private constructor - only builder can create instances
    private ApiConfiguration(Builder builder) {
        this.id = builder.id;
        this.serviceName = requireNonNull(builder.serviceName, "Service name is required");
        this.baseUrl = requireNonNull(builder.baseUrl, "Base URL is required");
        this.apiVersion = builder.apiVersion != null ? builder.apiVersion : "v1";
        this.timeoutMs = builder.timeoutMs > 0 ? builder.timeoutMs : 5000;
        this.retryCount = Math.max(0, builder.retryCount);
        this.sslEnabled = builder.sslEnabled;
        this.authToken = builder.authToken;
        this.userAgent = builder.userAgent != null ? builder.userAgent : "Spring-Boot-App/3.5";
        this.customHeaders = Collections.unmodifiableMap(
            builder.customHeaders != null ? builder.customHeaders : Collections.emptyMap()
        );
        
        validate();
    }
    
    // Static factory method
    public static Builder builder() {
        return new Builder();
    }
}

Fluent Builder Implementation

public static class Builder {
    private Long id;
    private String serviceName;
    private String baseUrl;
    private String apiVersion;
    private int timeoutMs = 5000;
    private int retryCount = 3;
    private boolean sslEnabled = true;
    private String authToken;
    private String userAgent;
    private Map<String, String> customHeaders = new HashMap<>();
    
    // Fluent interface methods
    public Builder serviceName(String serviceName) {
        this.serviceName = serviceName;
        return this;
    }
    
    public Builder baseUrl(String baseUrl) {
        this.baseUrl = baseUrl;
        return this;
    }
    
    public Builder apiVersion(String apiVersion) {
        this.apiVersion = apiVersion;
        return this;
    }
    
    public Builder timeout(Duration timeout) {
        this.timeoutMs = (int) timeout.toMillis();
        return this;
    }
    
    public Builder retryCount(int retryCount) {
        this.retryCount = retryCount;
        return this;
    }
    
    public Builder enableSsl(boolean enabled) {
        this.sslEnabled = enabled;
        return this;
    }
    
    public Builder authToken(String token) {
        this.authToken = token;
        return this;
    }
    
    public Builder userAgent(String userAgent) {
        this.userAgent = userAgent;
        return this;
    }
    
    public Builder addHeader(String key, String value) {
        this.customHeaders.put(key, value);
        return this;
    }
    
    public Builder headers(Map<String, String> headers) {
        this.customHeaders.putAll(headers);
        return this;
    }
    
    // Build method with validation
    public ApiConfiguration build() {
        validateBuilder();
        return new ApiConfiguration(this);
    }
    
    private void validateBuilder() {
        if (serviceName == null || serviceName.isBlank()) {
            throw new IllegalArgumentException("Service name cannot be null or blank");
        }
        
        if (baseUrl == null || !isValidUrl(baseUrl)) {
            throw new IllegalArgumentException("Base URL must be valid");
        }
        
        if (timeoutMs <= 0) {
            throw new IllegalArgumentException("Timeout must be positive");
        }
    }
}

Service Layer ile Integration

@Service
@Transactional
public class ApiConfigurationService {
    
    private final ApiConfigurationRepository repository;
    private final ConfigurationValidator validator;
    
    public ApiConfigurationService(
            ApiConfigurationRepository repository,
            ConfigurationValidator validator) {
        this.repository = repository;
        this.validator = validator;
    }
    
    public ApiConfiguration createConfiguration(ConfigurationRequest request) {
        // Builder pattern usage in service layer
        ApiConfiguration config = ApiConfiguration.builder()
            .serviceName(request.getServiceName())
            .baseUrl(request.getBaseUrl())
            .apiVersion(request.getApiVersion())
            .timeout(Duration.ofSeconds(request.getTimeoutSeconds()))
            .retryCount(request.getRetryCount())
            .enableSsl(request.isEnableSsl())
            .authToken(request.getAuthToken())
            .userAgent(buildUserAgent(request))
            .headers(request.getCustomHeaders())
            .build();
            
        // Additional business validation
        validator.validateConfiguration(config);
        
        return repository.save(config);
    }
    
    public ApiConfiguration createDevelopmentConfig(String serviceName) {
        // Pre-configured builder for development environment
        return ApiConfiguration.builder()
            .serviceName(serviceName)
            .baseUrl("http://localhost:8080")
            .apiVersion("v1")
            .timeout(Duration.ofSeconds(30))
            .retryCount(1)
            .enableSsl(false)
            .addHeader("Environment", "Development")
            .build();
    }
    
    public ApiConfiguration createProductionConfig(String serviceName, String baseUrl) {
        // Production-ready configuration with security defaults
        return ApiConfiguration.builder()
            .serviceName(serviceName)
            .baseUrl(baseUrl)
            .apiVersion("v2")
            .timeout(Duration.ofSeconds(10))
            .retryCount(5)
            .enableSsl(true)
            .addHeader("Accept", "application/json")
            .addHeader("Content-Type", "application/json")
            .build();
    }
    
    private String buildUserAgent(ConfigurationRequest request) {
        return String.format("%s/%s (Spring-Boot/3.5)", 
            request.getServiceName(), 
            request.getApiVersion());
    }
}

REST Controller Implementation

@RestController
@RequestMapping("/api/v1/configurations")
@Validated
public class ApiConfigurationController {
    
    private final ApiConfigurationService configurationService;
    
    public ApiConfigurationController(ApiConfigurationService configurationService) {
        this.configurationService = configurationService;
    }
    
    @PostMapping
    public ResponseEntity<ApiConfigurationDto> createConfiguration(
            @Valid @RequestBody ConfigurationRequest request) {
        
        try {
            ApiConfiguration config = configurationService.createConfiguration(request);
            ApiConfigurationDto dto = mapToDto(config);
            
            return ResponseEntity
                .status(HttpStatus.CREATED)
                .body(dto);
                
        } catch (IllegalArgumentException e) {
            return ResponseEntity.badRequest().build();
        }
    }
    
    @PostMapping("/quick-setup/{environment}")
    public ResponseEntity<ApiConfigurationDto> createQuickConfiguration(
            @PathVariable String environment,
            @RequestParam String serviceName,
            @RequestParam(required = false) String baseUrl) {
        
        ApiConfiguration config = switch (environment.toLowerCase()) {
            case "development" -> configurationService.createDevelopmentConfig(serviceName);
            case "production" -> {
                if (baseUrl == null) {
                    yield ResponseEntity.badRequest().build();
                }
                yield configurationService.createProductionConfig(serviceName, baseUrl);
            }
            default -> throw new IllegalArgumentException("Unsupported environment: " + environment);
        };
        
        return ResponseEntity.ok(mapToDto(config));
    }
}

// Type-safe stepwise builder implementation
public class TypeSafeApiConfigurationBuilder {
    
    public static ServiceNameStep newBuilder() {
        return new Steps();
    }
    
    public interface ServiceNameStep {
        BaseUrlStep serviceName(String serviceName);
    }
    
    public interface BaseUrlStep {
        OptionalStep baseUrl(String baseUrl);
    }
    
    public interface OptionalStep {
        OptionalStep apiVersion(String version);
        OptionalStep timeout(Duration timeout);
        OptionalStep retryCount(int count);

    }

        private String serviceName;
        private String baseUrl;

        @Override
        public BaseUrlStep serviceName(String serviceName) {

            return this;
        }
        
        @Override
        public OptionalStep baseUrl(String baseUrl) {
            this.baseUrl = baseUrl;
            return this;
        }
        
        // Implementation of optional methods...
        
        @Override
        public ApiConfiguration build() {
            return ApiConfiguration.builder()
                .serviceName(serviceName)
                .baseUrl(baseUrl)
                .apiVersion(apiVersion)
                .timeout(timeout)
                .retryCount(retryCount)
                .build();
        }
    }

// Custom validation with Builder pattern
@Component
public class ConfigurationValidator {
    
    private final Validator validator;
    
    public ConfigurationValidator(Validator validator) {
        this.validator = validator;
    }
    
    public void validateConfiguration(ApiConfiguration config) {
        Set<ConstraintViolation<ApiConfiguration>> violations = 
            validator.validate(config);
            
        if (!violations.isEmpty()) {
            String errorMessage = violations.stream()
                .map(ConstraintViolation::getMessage)
                .collect(Collectors.joining(", "));
            throw new ValidationException("Configuration validation failed: " + errorMessage);
        }
        
        // Custom business logic validation
        if (config.isSslEnabled() && !config.getBaseUrl().startsWith("https")) {
            throw new ValidationException("SSL enabled but URL is not HTTPS");
        }
    }
}

@Component
public class PooledApiConfigurationBuilder {

    public PooledApiConfigurationBuilder() {

            builderPool.returnObject(builder);
        }

// Conditional builder with lambda support
public static class ConditionalBuilder<T> implements FluentBuilder<T> {
    
    private final FluentBuilder<T> builder;
    
    public ConditionalBuilder(FluentBuilder<T> builder) {
        this.builder = builder;
    }
    
    public ConditionalBuilder<T> when(boolean condition, 
            Function<FluentBuilder<T>, FluentBuilder<T>> configurator) {
        if (condition) {

        }
        return this;
    }
    
    @Override
    public T build() {
        return builder.build();
    }
}

// Complete production-ready example
@RestController
@RequestMapping("/api/v1/configurations")
public class ProductionConfigurationController {
    
    @PostMapping("/batch")
    @Transactional
    public ResponseEntity<List<ApiConfigurationResponse>> createBatchConfigurations(
            @Valid @RequestBody BatchConfigurationRequest request) {
        
        List<ApiConfiguration> configurations = request.getConfigurations()
            .stream()
            .map(this::buildConfiguration)
            .collect(Collectors.toList());
            
        List<ApiConfiguration> savedConfigs = configurationService.saveAll(configurations);
        
        List<ApiConfigurationResponse> responses = savedConfigs.stream()
            .map(this::mapToResponse)
            .collect(Collectors.toList());
            
        return ResponseEntity.ok(responses);
    }
    
    private ApiConfiguration buildConfiguration(ConfigurationRequest request) {
        return ApiConfiguration.builder()
            .serviceName(request.getServiceName())
            .baseUrl(request.getBaseUrl())
            .apiVersion(Optional.ofNullable(request.getApiVersion()).orElse("v1"))
            .timeout(Duration.ofSeconds(
                Optional.ofNullable(request.getTimeoutSeconds()).orElse(5)))
            .retryCount(Optional.ofNullable(request.getRetryCount()).orElse(3))
            .enableSsl(Optional.ofNullable(request.getEnableSsl()).orElse(true))
            .authToken(request.getAuthToken())
            .headers(Optional.ofNullable(request.getCustomHeaders())
                .orElse(Collections.emptyMap()))
            .build();
    }
}
