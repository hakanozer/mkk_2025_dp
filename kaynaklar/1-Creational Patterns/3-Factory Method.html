Service Layer: Factory Injection

@Service
public class NotificationServiceManager {
    
    private final Map<String, NotificationFactory> factories;

    // Spring automatically injects all beans of type NotificationFactory into a Map
    public NotificationServiceManager(Map<String, NotificationFactory> factories) {
        this.factories = factories;
    }
    
    public String sendNotificationByType(String type, String recipient, String message) {
        NotificationFactory factory = factories.get(type + "Factory");
        if (factory == null) {
            throw new IllegalArgumentException("Invalid notification type: " + type);
        }
        NotificationService service = factory.createNotificationService();
        return service.sendNotification(recipient, message);
    }
}

            @RequestParam String type,

            return ResponseEntity.badRequest().body(e.getMessage());
        }
    }
}
