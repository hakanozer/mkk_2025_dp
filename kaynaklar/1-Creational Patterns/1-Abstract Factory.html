Service Layer Implementation

@Service
public class PaymentService {
    
    private final Map<String, PaymentProcessorFactory> factories;
    
    public PaymentService(Map<String, PaymentProcessorFactory> factories) {
        this.factories = factories;
    }
    
    public PaymentResult processPayment(PaymentRequest request) {
        // Factory Selection Strategy
        PaymentProcessorFactory factory = selectFactory(request.getPaymentType());
        
        // Create product family
        PaymentValidator validator = factory.createPaymentValidator();
        PaymentProcessor processor = factory.createPaymentProcessor();
        PaymentNotifier notifier = factory.createPaymentNotifier();
        
        // Business Logic
        if (!validator.validatePayment(request)) {
            throw new InvalidPaymentException("Payment validation failed");
        }
        
        PaymentResult result = processor.processPayment(request);
        notifier.sendNotification(result);
        
        return result;
    }
    
    private PaymentProcessorFactory selectFactory(PaymentType type) {
        return switch (type) {
            case CREDIT_CARD -> factories.get("creditCard");
            case DIGITAL_WALLET -> factories.get("digitalWallet");
            default -> throw new UnsupportedPaymentTypeException(type);
        };
    }
}

REST Controller

@RestController
@RequestMapping("/api/v1/payments")
@Validated
public class PaymentController {
    
    private final PaymentService paymentService;
    
    public PaymentController(PaymentService paymentService) {
        this.paymentService = paymentService;
    }
    
    @PostMapping("/process")
    public ResponseEntity<PaymentResult> processPayment(
            @Valid @RequestBody PaymentRequest request) {
        
        try {
            PaymentResult result = paymentService.processPayment(request);
            return ResponseEntity.ok(result);
        } catch (InvalidPaymentException e) {
            return ResponseEntity.badRequest().build();
        } catch (UnsupportedPaymentTypeException e) {
            return ResponseEntity.unprocessableEntity().build();
        }
    }
}
